§1 · Feature Architecture

Three-Layer Integration Architecture

The WHR module is a self-contained component that attaches to your existing dashboard without restructuring it. It reads from three data streams, runs the engineering model, and writes KPIs to the output layer.

INPUT LAYER
IoT sensors (Part 1)
DB / pipeline (Part 2)
User sidebar inputs
Default fallback values
→
PROCESSING LAYER
whr_engine.py
heat_calc() per source
scoring_model()
roi_calc()
→
OUTPUT LAYER
KPI cards (4 metrics)
Priority ranking table
ROI curves (12-year)
Radar / scatter charts

whr/engine.py — calculation logic, fully traceable

pythonwhr/engine.py
from dataclasses import dataclass
from typing import Optional
import numpy as np
from .config import CONST, EQUIPMENT, DEFAULTS, WEIGHTS


# ─── DATA CONTAINERS ──────────────────────────────────────────────
@dataclass
class SensorData:
    """Live or cached sensor readings from Part 1 IoT."""
    t_flue_in:      Optional[float] = None   # °C — thermocouple cheminée
    v_gn_flow:      Optional[float] = None   # Nm³/h — GN meter
    P_comp_meas:    Optional[float] = None   # kW — compressor energy meter
    t_GEG_in:       Optional[float] = None   # °C — GEG condenser inlet temp
    t_GEG_out:      Optional[float] = None   # °C — GEG condenser outlet temp
    t_cta_extract:  Optional[float] = None   # °C — CTA extract air temp
    is_realtime:    bool = False

@dataclass
class UserParams:
    """User-controlled parameters from dashboard sidebar."""
    t_flue_in:          float = DEFAULTS["t_flue_in"]
    t_flue_out_target:  float = DEFAULTS["t_flue_out_target"]
    eta_hx:             float = DEFAULTS["eta_hx"]
    eta_r_comp:         float = DEFAULTS["eta_r_comp"]
    p_gn:               float = DEFAULTS["p_gn"]
    p_elec:             float = DEFAULTS["p_elec"]
    capex_s1:           float = DEFAULTS["capex_s1"]
    capex_s2:           float = DEFAULTS["capex_s2"]
    capex_s3:           float = DEFAULTS["capex_s3"]

@dataclass
class WHRResult:
    """All outputs from a single engine.run() call."""
    # Powers [kW]
    Q_W1: float; Q_W2: float; Q_W3: float; Q_W5: float
    # Annual energy [MWh/yr]
    E_W1: float; E_W2: float; E_W3: float; E_W5: float
    E_total: float
    # CO2 [tCO2/yr]
    co2_W1: float; co2_W2: float; co2_W3: float
    co2_total: float
    # Financial [DT]
    savings_s1: float; savings_s2: float; savings_s3: float
    savings_total: float
    roi_s1: float; roi_s2: float; roi_s3: float
    roi_weighted: float
    # Scores
    scores: dict  # {source_id: {C1..C5, total}}
    data_source: str  # "realtime" | "static"
    warnings: list    # list of validation warnings


# ─── MAIN ENGINE CLASS ────────────────────────────────────────────
class WHREngine:

    def __init__(self, sensors: SensorData, params: UserParams):
        self.s = sensors
        self.p = params
        self.warnings = []

    def _resolve(self, sensor_val, default, name):
        """Use sensor if valid, else fallback to default with warning."""
        if sensor_val is not None:
            from .config import VALID_RANGES
            if name in VALID_RANGES:
                lo, hi = VALID_RANGES[name]
                if not (lo <= sensor_val <= hi):
                    self.warnings.append(
                        f"{name}={sensor_val} out of range [{lo},{hi}] → using default {default}"
                    )
                    return default, "default"
            return sensor_val, "sensor"
        self.warnings.append(f"{name}: no sensor → using default {default}")
        return default, "default"

    # ─ EQ-1 + EQ-3: W1 Flue gas from boilers ─────────────────────
    def calc_W1(self) -> tuple:
        # Step 1: resolve T_in from sensor or default
        T_in, src = self._resolve(self.s.t_flue_in, self.p.t_flue_in, "t_flue_in")
        T_out = self.p.t_flue_out_target   # user-defined constraint (≥130°C acid dew)

        # Step 2: derive flue gas flow via GN meter (EQ-3)
        if self.s.v_gn_flow is not None:
            v_gn_nm3h = self.s.v_gn_flow
        else:
            # derive from nominal: P_nom × τ_boiler / η / PCI × 3600
            v_gn_nm3h = (
                EQUIPMENT["P_boiler_nom"] * EQUIPMENT["tau_boiler"]
                / EQUIPMENT["eta_boiler"]
                / CONST["PCI_GN"] * 3600
            )  # Nm³/h

        v_flue_m3s = (
            v_gn_nm3h * CONST["r_stoich_GN"] / 3600
        )  # Nm³/s — EQ-3

        # Step 3: sensible heat recovery (EQ-1 volumetric form)
        Q_kw = (
            v_flue_m3s * CONST["Cp_flue_vol"]
            * (T_in - T_out) * self.p.eta_hx
        )
        E_mwh = Q_kw * EQUIPMENT["h_boiler"] / 1000
        return round(Q_kw, 1), round(E_mwh, 1)

    # ─ EQ-4: W2 Compressor heat recovery ─────────────────────────
    def calc_W2(self) -> tuple:
        # Resolve load factor from live energy meter or audit value
        if self.s.P_comp_meas is not None:
            tau = self.s.P_comp_meas / EQUIPMENT["P_comp_nom"]
            tau = max(0.05, min(1.0, tau))  # clamp
        else:
            tau = EQUIPMENT["tau_comp"]

        Q_kw = (
            EQUIPMENT["P_comp_nom"] * tau
            * 0.80                   # η_thermal loss (1 - isentropic η)
            * self.p.eta_r_comp    # WHR kit efficiency
        )
        E_mwh = Q_kw * EQUIPMENT["h_comp"] / 1000
        return round(Q_kw, 1), round(E_mwh, 1)

    # ─ EQ-5: W3 GEG desuperheating ────────────────────────────────
    def calc_W3(self) -> tuple:
        Q_cond = EQUIPMENT["Q_GEG_frig"] + EQUIPMENT["P_GEG_comp"]
        Q_kw   = 0.12 * Q_cond       # desuperheating fraction
        E_mwh  = Q_kw * EQUIPMENT["h_GEG"] / 1000
        return round(Q_kw, 1), round(E_mwh, 1)

    # ─ EQ-2: W5 Boiler purge flash ────────────────────────────────
    def calc_W5(self) -> tuple:
        steam_kgs = (
            EQUIPMENT["P_boiler_nom"] * EQUIPMENT["tau_boiler"]
            / CONST["h_fg_5bar"]
        )  # kg/s
        m_purge = 0.04 * steam_kgs   # 4% typical purge rate
        Q_kw    = m_purge * CONST["h_fg_5bar"]
        E_mwh   = Q_kw * EQUIPMENT["h_boiler"] / 1000
        return round(Q_kw, 1), round(E_mwh, 1)

    # ─ CO2 and financial calculations ─────────────────────────────
    def calc_co2(self, E_mwh, energy_type="GN") -> float:
        factor = CONST["f_CO2_GN"] if energy_type=="GN" else CONST["f_CO2_elec"]
        return round(E_mwh * factor, 1)  # tCO2/yr

    def calc_savings(self, E_mwh, energy_type="GN") -> float:
        price = self.p.p_gn if energy_type=="GN" else self.p.p_elec
        return round(E_mwh * price, 0)    # DT/yr

    # ─ Scoring model ──────────────────────────────────────────────
    def calc_scores(self, results_map) -> dict:
        """
        results_map: {
          "W1": {"E": 174, "co2": 35.1, "feasibility": 10, "capex": 45000, "roi": 2.7},
          ...
        }
        """
        def score_energy(e):
            if e >= 500:  return 9.5
            if e >= 300:  return 7.5
            if e >= 150:  return 5.5
            if e >= 50:   return 3.5
            return 1.5

        def score_co2(c):
            if c >= 100:  return 9.5
            if c >= 60:   return 7.5
            if c >= 30:   return 5.5
            if c >= 10:   return 3.5
            return 1.5

        def score_capex(c):
            if c < 20000:   return 9.5   # quick win
            if c < 50000:   return 7.5
            if c < 100000:  return 5.5
            if c < 200000:  return 3.5
            return 1.5

        def score_roi(r):
            if r < 1:    return 9.5
            if r < 2:    return 7.5
            if r < 3:    return 5.5
            if r < 5:    return 3.5
            return 1.5

        w = WEIGHTS
        scored = {}
        for src_id, d in results_map.items():
            c1 = score_energy(d["E"])
            c2 = score_co2(d["co2"])
            c3 = d.get("feasibility", 5)  # set manually per source
            c4 = score_capex(d["capex"])
            c5 = score_roi(d["roi"])
            total = (
                w["C1_energy"]      * c1 +
                w["C2_co2"]         * c2 +
                w["C3_feasibility"] * c3 +
                w["C4_capex"]       * c4 +
                w["C5_roi"]         * c5
            )
            scored[src_id] = {
                "C1": c1, "C2": c2, "C3": c3,
                "C4": c4, "C5": c5,
                "total": round(total, 2)
            }
        return scored

    # ─ MAIN ENTRY POINT ───────────────────────────────────────────
    def run(self) -> WHRResult:
        # 1. Compute heat powers and annual energies
        Q_W1, E_W1 = self.calc_W1()
        Q_W2, E_W2 = self.calc_W2()
        Q_W3, E_W3 = self.calc_W3()
        Q_W5, E_W5 = self.calc_W5()

        # 2. CO2 reductions
        co2_W1 = self.calc_co2(E_W1, "GN")
        co2_W2 = self.calc_co2(E_W2, "elec")   # replaces electric heating
        co2_W3 = self.calc_co2(E_W3, "elec")

        # 3. Savings and ROI per scenario
        sav_s1 = self.calc_savings(E_W1, "GN")
        sav_s2 = self.calc_savings(E_W2, "elec")
        sav_s3 = self.calc_savings(E_W3, "elec")
        roi_s1 = round(self.p.capex_s1 / sav_s1, 2) if sav_s1 > 0 else float("inf")
        roi_s2 = round(self.p.capex_s2 / sav_s2, 2) if sav_s2 > 0 else float("inf")
        roi_s3 = round(self.p.capex_s3 / sav_s3, 2) if sav_s3 > 0 else float("inf")
        capex_total = self.p.capex_s1 + self.p.capex_s2 + self.p.capex_s3
        sav_total   = sav_s1 + sav_s2 + sav_s3

        # 4. Scoring model
        src_map = {
            "W1": {"E": E_W1, "co2": co2_W1, "feasibility": 10.0, "capex": self.p.capex_s1, "roi": roi_s1},
            "W2": {"E": E_W2, "co2": co2_W2, "feasibility":  9.0, "capex": self.p.capex_s2, "roi": roi_s2},
            "W3": {"E": E_W3, "co2": co2_W3, "feasibility":  7.0, "capex": self.p.capex_s3, "roi": roi_s3},
        }
        scores = self.calc_scores(src_map)

        return WHRResult(
            Q_W1=Q_W1, Q_W2=Q_W2, Q_W3=Q_W3, Q_W5=Q_W5,
            E_W1=E_W1, E_W2=E_W2, E_W3=E_W3, E_W5=E_W5,
            E_total=round(E_W1+E_W2+E_W3, 1),
            co2_W1=co2_W1, co2_W2=co2_W2, co2_W3=co2_W3,
            co2_total=round(co2_W1+co2_W2+co2_W3, 1),
            savings_s1=sav_s1, savings_s2=sav_s2, savings_s3=sav_s3,
            savings_total=sav_total,
            roi_s1=roi_s1, roi_s2=roi_s2, roi_s3=roi_s3,
            roi_weighted=round(capex_total/sav_total, 2) if sav_total > 0 else 99,
            scores=scores,
            data_source="realtime" if self.s.is_realtime else "static",
            warnings=self.warnings
        )

def render_sidebar() -> UserParams:
    st.sidebar.markdown("### ♻ WHR Parameters")

    with st.sidebar.expander("Energy Prices", expanded=True):

        eta_r     = st.slider("WHR kit η (%)",         50,  80,  65) / 100

    with st.sidebar.expander("CAPEX Estimates (DT)"):
        c1 = st.number_input("S1 Economizer",    0, 200000, 45000,  1000)
        c2 = st.number_input("S2 Compressor WHR", 0, 100000, 19000,  1000)
        c3 = st.number_input("S3 GEG Desuperheater", 0, 400000, 110000, 5000)

    with st.sidebar.expander("Scoring Weights"):
        w1 = st.slider("C1 Energy",      0.0, 0.5, 0.30, 0.05)
        w2 = st.slider("C2 CO2",         0.0, 0.5, 0.20, 0.05)
        w3 = st.slider("C3 Feasibility", 0.0, 0.5, 0.20, 0.05)
        w4 = st.slider("C4 CAPEX",       0.0, 0.5, 0.15, 0.05)

    return UserParams(

        p_gn=p_gn, p_elec=p_elec,

Per scenario and weighted total. Line chart over 12 years.

import streamlit as st
import pandas as pd
import plotly.express as px
import plotly.graph_objects as go
import numpy as np
from .engine import WHREngine
from .data_bridge import fetch_realtime_sensors
from .ui import render_sidebar

def render():
    # ── 0. Sidebar inputs + sensor fetch ──────────────────────────
    params  = render_sidebar()
    sensors = fetch_realtime_sensors()

    # ── 1. Run engine ─────────────────────────────────────────────
    res = WHREngine(sensors, params).run()

    # ── 2. Status banner ──────────────────────────────────────────
    mode_color = "🟢" if res.data_source=="realtime" else "🟡"
    st.caption(f"{mode_color} Mode: {res.data_source.upper()} | Warnings: {len(res.warnings)}")
    if res.warnings:
        with st.expander(f"⚠ {len(res.warnings)} data warning(s)"):
            for w in res.warnings: st.warning(w)

    # ── 3. Top KPI row ────────────────────────────────────────────
    k1, k2, k3, k4 = st.columns(4)
    k1.metric("♻ Recoverable Energy", f"{res.E_total:.0f} MWh/yr",
              delta=f"vs 0 (baseline)")
    k2.metric("🌱 CO₂ Avoided",        f"{res.co2_total:.1f} tCO₂/yr")
    k3.metric("💰 Annual Savings",     f"{res.savings_total:,.0f} DT/yr")
    k4.metric("📈 Weighted ROI",       f"{res.roi_weighted:.1f} yrs")

    st.divider()

    # ── 4. Energy breakdown chart ─────────────────────────────────
    col_a, col_b = st.columns(2)

    with col_a:
        st.subheader("Energy by Source (MWh/yr)")
        df_e = pd.DataFrame({
            "Source": ["W1 Flue gas", "W2 Compressor", "W3 GEG Cond."],
            "E_MWh": [res.E_W1, res.E_W2, res.E_W3],
            "Scenario": ["S1", "S2", "S3"],
            "Color": ["#f59e0b", "#10b981", "#00d4ff"]
        })
        fig_e = px.bar(df_e, x="E_MWh", y="Source", orientation="h",
                       color="Scenario", text="E_MWh",
                       color_discrete_map={"S1":"#f59e0b","S2":"#10b981","S3":"#00d4ff"})
        fig_e.update_traces(texttemplate="%{x:.0f} MWh", textposition="outside")
        fig_e.update_layout(height=240, margin=dict(l=0,r=0,t=20,b=0),
                            plot_bgcolor="rgba(0,0,0,0)", paper_bgcolor="rgba(0,0,0,0)")
        st.plotly_chart(fig_e, use_container_width=True)

    with col_b:
        st.subheader("ROI Cumulative Cash-flow (12 yr)")
        years = np.arange(0, 13)
        fig_roi = go.Figure()
        for label, capex, sav, col in [
            ("S1 Economizer",   params.capex_s1, res.savings_s1, "#f59e0b"),
            ("S2 Compressor",  params.capex_s2, res.savings_s2, "#10b981"),
            ("S3 GEG Cond.",    params.capex_s3, res.savings_s3, "#00d4ff"),
        ]:
            cum = [-capex + sav*y for y in years]
            fig_roi.add_trace(go.Scatter(x=years, y=cum, name=label,
                               line=dict(color=col, width=2)))
        fig_roi.add_hline(y=0, line_dash="dash", line_color="gray")
        fig_roi.update_layout(height=240, xaxis_title="Year", yaxis_title="DT",
                             margin=dict(l=0,r=0,t=20,b=0),
                             plot_bgcolor="rgba(0,0,0,0)", paper_bgcolor="rgba(0,0,0,0)")
        st.plotly_chart(fig_roi, use_container_width=True)

§8 · Error Handling & Validation

Validation & Fault Tolerance

Sensor out-of-range

Checked in _resolve() against VALID_RANGES. Out-of-range value → replaced by default, warning added to res.warnings, shown in expandable banner. Engine always completes.

# Example: t_flue_in = 820°C → sensor fault
# valid range (100, 350) → uses default 190°C
# warning: "t_flue_in=820 out of range [100,350]"

Missing sensor (None)

All SensorData fields are Optional. If None, _resolve() falls back to config default silently in static mode, warns in realtime mode. No crash possible.

# If v_gn_flow is None:
# → derives from P_boiler_nom × tau_boiler
# → warns: "v_gn_flow: no sensor → using derived"

Division by zero (ROI)

Savings can be zero if energy price is zero. Protected with explicit check before division. Returns float("inf") → displayed as "N/A" in the UI.

roi = (capex / savings
       if savings > 0
       else float("inf"))
display = (f"{roi:.1f} yrs"
           if roi < 99
           else "N/A")

Physical constraint: acid dew point

If user sets T_out < 110°C, show a hard warning. Do not block calculation — engineer may have a reason — but flag clearly.

if params.t_flue_out_target < 110:
    st.error(
        "⚠ T_out below acid dew point (110°C)."
        " Risk of sulfuric condensation → corrosion."
    )

Scoring weight sum ≠ 1.0

C5 weight is auto-computed as 1.0 - w1 - w2 - w3 - w4. If user somehow forces a negative C5, clamp to 0 and warn. Normalization always enforced.

w5 = max(0.0, 1.0 - w1 - w2 - w3 - w4)
# always Σwi = 1.0

Data bridge connection failure

fetch_realtime_sensors() is wrapped in try/except. Any exception → returns empty SensorData() (all None). Engine runs in static mode. User sees 🟡 STATIC banner. Never crashes the dashboard.

except Exception:
    return SensorData()
# → graceful degradation to static mode

Design principle: the engine never raises exceptions to the UI layer. All errors are caught, logged to res.warnings, and displayed as non-blocking banners. The dashboard always shows a result — even if all sensors are down.

from whr.engine import WHREngine, SensorData, UserParams

def test_static_mode_totals():
    r = WHREngine(SensorData(), UserParams()).run()
    assert 800 < r.E_total < 950,  f"E_total={r.E_total} unexpected"
    assert 150 < r.co2_total < 220, f"CO2={r.co2_total} unexpected"
    assert 1.5 < r.roi_weighted < 3, f"ROI={r.roi_weighted} unexpected"
    assert len(r.warnings) > 0,      "Should warn about missing sensors"

def test_sensor_override():
    # Real T_in = 160°C → Q_W1 should drop ~30%
    r_default = WHREngine(SensorData(), UserParams()).run()
    r_sensor  = WHREngine(SensorData(t_flue_in=160, is_realtime=True),
                          UserParams()).run()
    assert r_sensor.E_W1 < r_default.E_W1, "Sensor override did not reduce E_W1"

def test_zero_savings_guard():
    # price=0 should not crash
    p = UserParams(p_gn=0, p_elec=0)
    r = WHREngine(SensorData(), p).run()
    assert r.roi_weighted == 99, "Should return sentinel 99 not crash"
